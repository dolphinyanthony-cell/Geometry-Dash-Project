function drawBackground(){ // dark base let grad = ctx.createLinearGradient(0,0,0,canvas.height); grad.addColorStop(0, "#05070c"); grad.addColorStop(1, "#000000"); ctx.fillStyle = grad; ctx.fillRect(0,0,canvas.width,canvas.height); // subtle leafy silhouette layer (no images, just shapes) ctx.fillStyle = "rgba(0,255,140,0.05)"; let t = performance.now() * 0.0003; for(let i = 0; i < 25; i++){ let x = (i * 180 + Math.sin(t + i) * 40) % (canvas.width + 200); let y = (i * 90) % canvas.height; drawLeaf(x, y, 30 + (i % 3) * 10); } // black fade overlay for “menu vibe” let fade = ctx.createLinearGradient(0,0,0,canvas.height); fade.addColorStop(0, "rgba(0,0,0,0.2)"); fade.addColorStop(1, "rgba(0,0,0,0.85)"); ctx.fillStyle = fade; ctx.fillRect(0,0,canvas.width,canvas.height); // ground glow line (GD style) ctx.strokeStyle = "rgba(0,255,200,0.25)"; ctx.lineWidth = 2; ctx.beginPath(); ctx.moveTo(0, canvas.height - 100); ctx.lineTo(canvas.width, canvas.height - 100); ctx.stroke(); }
